from pathlib import Path
import duckdb

# Project paths
project_root = Path(r"C:\Users\Owner\Desktop\Isha Healthcare project")
raw_dir = project_root / "data" / "raw" / "synthea_sample_data_csv_latest"
db_path = project_root / "warehouse" / "healthcare.duckdb"

# Connect to DuckDB
con = duckdb.connect(str(db_path))

# Files we want first
tables = [
    "claims",
    "claims_transactions",
    "patients",
    "encounters",
    "payers",
    "providers",
    "organizations",
    "procedures",
    "conditions",
]

for table in tables:
    csv_path = raw_dir / f"{table}.csv"
    print(f"Loading {table} from {csv_path}...")
    con.execute(f"""
        CREATE OR REPLACE TABLE {table} AS
        SELECT *
        FROM read_csv_auto('{csv_path.as_posix()}', header=True);
    """)

print("\nLoaded tables successfully.\n")

# Quick sanity check
for table in tables:
    count = con.execute(f"SELECT COUNT(*) FROM {table}").fetchone()[0]
    print(f"{table}: {count} rows")

con.close()
